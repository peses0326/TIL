x = int(input())


def ACM_hotel(h, w, n):
    # n번째 손님 h층 w객실
    floor = n % h  # 나머지 값이 층
    room_count = n // h + 1 # 몫이 호수

    if floor == 0:
        floor = h
        room_count -= 1
    if h == n:
        floor = h
        room_count = 1
    room = floor * 100 + room_count
    return room


for i in range(x):
    H, W, N = list(map(int, input().split()))
    print(ACM_hotel(H, W, N))
